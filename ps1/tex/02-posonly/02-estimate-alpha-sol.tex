    $$
    \begin{aligned}
        h(x^{(i)}) &\approx p(y^{(i)}=1 \mid x^{(i)}) \\
        &= \frac{p(y^{(i)}=1, x^{(i)})}{p(x^{(i)})}\\
        &\approx \frac{p(y^{(i)}=1, x^{(i)})}{p(t^{(i)} = 1, x^{(i)})} \\
        &= \alpha 
    \end{aligned}
    $$
